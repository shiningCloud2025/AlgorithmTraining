  public int lengthOfLongestSubstring(String s) {
        int res = 0;
        Set<Integer> set = new HashSet();
        char[] cs = s.toCharArray();
        for(int l=0,r=0;r<cs.length;r++){
            while(l<r&&set.contains(cs[r])){
                set.remove(cs[l++]);
            }
            set.add(cs[r]);
            res = Math.max(res,r-l+1);
        }
        return res;
    }
